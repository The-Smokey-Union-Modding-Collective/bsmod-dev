on_actions = {

	on_startup = {

		effect = {
			every_country = { # Add 100% compliance to every state that isn't owned by ITA or JAP
				limit = {
					NOT = {
						OR = {
							original_tag = ITA
							original_tag = JAP
						}
					}
					any_owned_state = {
						has_resistance = yes
					}
				}
				every_controlled_state = {
					set_compliance = 100
				}
			}
		}
		
	}
}
