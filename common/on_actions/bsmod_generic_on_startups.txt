on_actions = {
	on_startup = {
		effect = {
			every_country = { # Land Doctrine cost reduction
				if = {
					limit = {
						has_tech = mobile_warfare
					}
					add_doctrine_cost_reduction = {
						name = STARTING_COST_REDUCTION
						cost_reduction = 0.5
						uses = 1
						category = first_land_doctrine
					}
				}
				else_if = {
					limit = {
						has_tech = superior_firepower
					}
					add_doctrine_cost_reduction = {
						name = STARTING_COST_REDUCTION
						cost_reduction = 0.5
						uses = 1
						category = first_land_doctrine
					}
				}
				else_if = {
					limit = {
						has_tech = trench_warfare
					}
					add_doctrine_cost_reduction = {
						name = STARTING_COST_REDUCTION
						cost_reduction = 0.5
						uses = 1
						category = first_land_doctrine
					}
				}
				else_if = {
					limit = {
						has_tech = mass_assault
					}
					add_doctrine_cost_reduction = {
						name = STARTING_COST_REDUCTION
						cost_reduction = 0.5
						uses = 1
						category = first_land_doctrine
					}
				}
				else_if = { # handles the case where there is no current land doctrine
					limit = {
						land_doctrine_level < 1
					}
					add_doctrine_cost_reduction = {
						name = STARTING_COST_REDUCTION
						cost_reduction = 0.5
						uses = 1
						category = first_land_doctrine
					}
				}
			}
			every_country = { # Add 100% compliance to every state that isn't owned by ITA or JAP
				limit = {
					NOT = {
						OR = {
							original_tag = ITA
							original_tag = JAP
						}
					}
					OR = {
						original_tag = ENG
						original_tag = FRA
					}
					any_owned_state = {
						has_resistance = yes
					}
				}
				every_controlled_state = {
					set_compliance = 100
				}
			}
		}
	}
	on_startup = {
		effect = {
			every_country = {
				if = {
					limit = {
						NOT = {
							is_ai = yes
						}
					}
					country_event = { days = 0 id = tank_explanataions.1 }
				}
			}
		}
	}
	
	on_startup = {
		effect = {
			GER = {
				country_event = { days = 0 id = germany.1044 }
			}
		}
	}
}
